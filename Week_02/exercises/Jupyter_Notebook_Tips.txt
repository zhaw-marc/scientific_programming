#---------------------------------
# Handling
#---------------------------------
- run all codecells
    Menu: Kernel > Restart & Run All  (or Cell > Run All to keep kernel state)

- run single codecell
    Shift + Enter   -> run cell and move to next cell
    Ctrl  + Enter   -> run cell and stay on same cell
    Alt   + Enter   -> run cell and insert new cell below

- clear outputs
    Menu: Cell > All Output > Clear  (clears all outputs in the notebook)
    Menu: Cell > Current Outputs > Clear  (clears only the selected cell)

- restart
    Menu: Kernel > Restart (clears all variables, does NOT re-run cells)
    Menu: Kernel > Restart & Clear Output (also clears all cell outputs)
    Menu: Kernel > Restart & Run All (restarts and re-runs the entire notebook)

- define new short cuts
    Menu: Help > Keyboard Shortcuts (view all shortcuts)
    Menu: Help > Edit Keyboard Shortcuts (add or override shortcuts in JSON format)

#---------------------------------
# Show package version
#---------------------------------
# Works for any package that exposes __version__
import numpy as np
np.__version__          # returns the version string, e.g. '1.26.4'

# Alternative: works even when __version__ is not defined
import importlib.metadata
importlib.metadata.version("numpy")   # e.g. '1.26.4'

#---------------------------------
# Getting help
#---------------------------------
# Append ? to open the docstring in a pop-up pane at the bottom
np.random.normal?

# Use help() for a plain-text printout of the full docstring
help(np.random.normal)

# Append ?? to also show the source code (when available)
np.random.normal??

# Tab-completion: type np.random. and press Tab to browse available functions

#---------------------------------
# Magic commands
#---------------------------------
%pwd                        # print current working directory

x = 5
y = 6
z = x + y

%who                        # list all variables currently defined in the kernel
%whos                       # like %who but also shows type and value

%time sum(range(1000))      # measure wall time for a single run
%timeit sum(range(1000))    # measure average time over many runs (more accurate)

%hist                       # show command history of the current session
%magic                      # open full documentation for all magic commands

# Cell magic (prefix with %% instead of % and applies to the whole cell):
# %%time   -> time the entire cell
# %%timeit -> benchmark the entire cell

# Display Multiple Outputs in Code Cell
#--------------------------------------
# By default Jupyter only shows the last expression in a cell.
# Setting ast_node_interactivity = "all" makes every expression print its result.

from IPython.core.interactiveshell import InteractiveShell
import seaborn as sns

iris = sns.load_dataset('iris')

InteractiveShell.ast_node_interactivity = "all"  # options: "all", "last", "last_expr", "none"

iris.head(3)  # Output is displayed
iris.tail(3)  # Output is displayed
